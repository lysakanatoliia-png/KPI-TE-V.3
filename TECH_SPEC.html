3. SQL — повні CREATE TABLE скрипти

SQL install.sql — запустити один раз при деплої

-- ============================================== -- KPI V3 · Database Schema · install.sql -- Запустити через phpMyAdmin або mysql CLI -- ============================================== CREATE DATABASE IF NOT EXISTS te_kpi CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci; USE te_kpi; -- ---- ROOMS ---- CREATE TABLE IF NOT EXISTS rooms ( room_code VARCHAR(32) NOT NULL PRIMARY KEY, room_name VARCHAR(128) NOT NULL, is_admin TINYINT(1) NOT NULL DEFAULT 0, active TINYINT(1) NOT NULL DEFAULT 1 ) ENGINE=InnoDB; -- ---- SLOTS ---- CREATE TABLE IF NOT EXISTS slots ( slot_code VARCHAR(32) NOT NULL PRIMARY KEY, slot_label VARCHAR(64) NOT NULL, slot_short VARCHAR(32) DEFAULT NULL, sort_order INT NOT NULL DEFAULT 0, active TINYINT(1) NOT NULL DEFAULT 1 ) ENGINE=InnoDB; -- ---- STAFF ---- CREATE TABLE IF NOT EXISTS staff ( staff_id VARCHAR(64) NOT NULL PRIMARY KEY, staff_name VARCHAR(128) NOT NULL, role VARCHAR(64) DEFAULT NULL, rooms VARCHAR(256) DEFAULT NULL, -- "Pink,Green" active TINYINT(1) NOT NULL DEFAULT 1 ) ENGINE=InnoDB; -- ---- INDICATORS ---- CREATE TABLE IF NOT EXISTS indicators ( id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, room_code VARCHAR(32) NOT NULL DEFAULT '*', -- '*' = для всіх кімнат slot_code VARCHAR(32) NOT NULL DEFAULT '*', -- '*' = для всіх слотів scope ENUM('Team','Individual') NOT NULL DEFAULT 'Team', category VARCHAR(128) NOT NULL, indicator VARCHAR(512) NOT NULL, weight DECIMAL(5,2) NOT NULL DEFAULT 1.00, sort_order INT NOT NULL DEFAULT 0, active TINYINT(1) NOT NULL DEFAULT 1 ) ENGINE=InnoDB; -- ---- KPI ENTRIES (головна таблиця) ---- CREATE TABLE IF NOT EXISTS kpi_entries ( id INT NOT NULL AUTO_INCREMENT PRIMARY KEY, batch_id VARCHAR(64) NOT NULL, entry_date DATE NOT NULL, room_code VARCHAR(32) NOT NULL, slot_code VARCHAR(32) NOT NULL, staff_id VARCHAR(64) NOT NULL, scope ENUM('Team','Individual') NOT NULL, category VARCHAR(128) NOT NULL, indicator VARCHAR(512) NOT NULL, check_value TINYINT(1) NOT NULL DEFAULT 1, -- 1=yes, 0=no weight DECIMAL(5,2) NOT NULL DEFAULT 1.00, earned_points DECIMAL(8,2) NOT NULL DEFAULT 0.00, possible_points DECIMAL(8,2) NOT NULL DEFAULT 0.00, comment TEXT DEFAULT NULL, submitted_by VARCHAR(64) DEFAULT NULL, present_staff VARCHAR(512) DEFAULT NULL, -- "st_tamar,st_alla" для team-рядків created_at DATETIME NOT NULL DEFAULT NOW(), -- Дедублікація: один запис = один batch + staff + indicator UNIQUE KEY uq_entry (batch_id, staff_id, indicator(200)), -- Індекси для швидких запитів KEY idx_date (entry_date), KEY idx_room_date (room_code, entry_date), KEY idx_staff_date (staff_id, entry_date), KEY idx_batch (batch_id), KEY idx_violations (check_value, entry_date) ) ENGINE=InnoDB;

4. api/db.php — підключення до бази даних

PHP api/db.php

<?php declare(strict_types=1); require_once __DIR__.'/config.env.php'; date_default_timezone_set(TZ); function getDB(): PDO { static $pdo = null; if ($pdo === null) { $dsn = "mysql:host=".DB_HOST.";dbname=".DB_NAME.";charset=".DB_CHARSET; $pdo = new PDO($dsn, DB_USER, DB_PASS, [ PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION, PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC, PDO::ATTR_EMULATE_PREPARES => false, ]); } return $pdo; } function jsonOut(array $data, int $status = 200): void { http_response_code($status); header('Content-Type: application/json; charset=utf-8'); header('Access-Control-Allow-Origin: *'); echo json_encode($data, JSON_UNESCAPED_UNICODE | JSON_UNESCAPED_SLASHES); exit; } function jsonError(string $msg, int $status = 400): void { jsonOut(['ok' => false, 'error' => $msg], $status); } function getInput(): array { $raw = file_get_contents('php://input'); $data = json_decode($raw, true); if (!is_array($data)) jsonError('Invalid JSON body'); return $data; } function generateBatchId(): string { return 'B-' . round(microtime(true) * 1000); } function sanitizeDate(string $d): string { // Приймає MM-DD-YYYY або YYYY-MM-DD → повертає YYYY-MM-DD для MySQL if (preg_match('/^(\d{2})-(\d{2})-(\d{4})$/', $d, $m)) { return "{$m[3]}-{$m[1]}-{$m[2]}"; } if (preg_match('/^\d{4}-\d{2}-\d{2}$/', $d)) return $d; jsonError("Invalid date format: {$d}"); }

5. api/kpi.php — збереження KPI

PHP api/kpi.php

<?php declare(strict_types=1); require_once __DIR__.'/db.php'; $action = $_GET['action'] ?? ''; match($action) { 'saveTeam' => saveTeam(), 'saveIndividual' => saveIndividual(), 'checkDuplicate' => checkDuplicate(), default => jsonError('Unknown action', 404) }; /* ==================================================== saveTeam — зберегти командні індикатори Для кожного стафу з present_staff_ids → окремий рядок ==================================================== */ function saveTeam(): void { $in = getInput(); $db = getDB(); $batchId = $in['batch_id'] ?? jsonError('Missing batch_id'); $date = sanitizeDate($in['date'] ?? jsonError('Missing date')); $room = trim($in['room_code'] ?? jsonError('Missing room_code')); $slot = trim($in['slot_code'] ?? jsonError('Missing slot_code')); $staffIds= $in['present_staff_ids'] ?? []; $by = trim($in['submitted_by'] ?? 'ui'); $items = $in['items'] ?? []; $presentStr = implode(',', $staffIds); if (!$items || !$staffIds) jsonError('Empty items or staff list'); // Завантажити ваги індикаторів з БД $weights = loadIndicatorWeights($db, $room, $slot, 'Team'); $sql = "INSERT IGNORE INTO kpi_entries (batch_id, entry_date, room_code, slot_code, staff_id, scope, category, indicator, check_value, weight, earned_points, possible_points, comment, submitted_by, present_staff) VALUES (?,?,?,?,?, ?,?,?,?,?, ?,?,?,?,?)"; $stmt = $db->prepare($sql); $saved = 0; $db->beginTransaction(); try { foreach ($items as $item) { $ind = trim($item['indicator'] ?? ''); $cat = trim($item['category'] ?? ''); $check = (strtolower($item['check'] ?? 'yes') === 'yes') ? 1 : 0; $comment = trim($item['comment'] ?? '') ?: null; $w = $weights[strtolower($ind)] ?? 1.0; $earned = $check ? $w : 0.0; foreach ($staffIds as $staffId) { $stmt->execute([ $batchId, $date, $room, $slot, $staffId, 'Team', $cat, $ind, $check, $w, $earned, $w, $comment, $by, $presentStr ]); $saved += $stmt->rowCount(); } } $db->commit(); } catch (Throwable $e) { $db->rollBack(); jsonError('DB error: ' . $e->getMessage(), 500); } jsonOut(['ok' => true, 'saved' => $saved]); } /* ==================================================== saveIndividual — зберегти індивідуальні індикатори ==================================================== */ function saveIndividual(): void { $in = getInput(); $db = getDB(); $batchId = $in['batch_id'] ?? jsonError('Missing batch_id'); $date = sanitizeDate($in['date'] ?? jsonError('Missing date')); $room = trim($in['room_code'] ?? jsonError('Missing room_code')); $slot = trim($in['slot_code'] ?? jsonError('Missing slot_code')); $staffId = trim($in['staff_id'] ?? jsonError('Missing staff_id')); $by = trim($in['submitted_by'] ?? 'ui'); $items = $in['items'] ?? []; if (!$items) jsonError('Empty items'); $weights = loadIndicatorWeights($db, $room, $slot, 'Individual'); $sql = "INSERT IGNORE INTO kpi_entries (batch_id, entry_date, room_code, slot_code, staff_id, scope, category, indicator, check_value, weight, earned_points, possible_points, comment, submitted_by) VALUES (?,?,?,?,?, ?,?,?,?,?, ?,?,?,?)"; $stmt = $db->prepare($sql); $saved = 0; $db->beginTransaction(); try { foreach ($items as $item) { $ind = trim($item['indicator'] ?? ''); $cat = trim($item['category'] ?? ''); $check = (strtolower($item['check'] ?? 'yes') === 'yes') ? 1 : 0; $comment = trim($item['comment'] ?? '') ?: null; $w = $weights[strtolower($ind)] ?? 1.0; $earned = $check ? $w : 0.0; $stmt->execute([ $batchId, $date, $room, $slot, $staffId, 'Individual', $cat, $ind, $check, $w, $earned, $w, $comment, $by ]); $saved += $stmt->rowCount(); } $db->commit(); } catch (Throwable $e) { $db->rollBack(); jsonError('DB error: ' . $e->getMessage(), 500); } jsonOut(['ok' => true, 'saved' => $saved]); } /* ==================================================== checkDuplicate — перевірка перед збереженням ==================================================== */ function checkDuplicate(): void { $batchId = trim($_GET['batch_id'] ?? ''); $staffId = trim($_GET['staff_id'] ?? ''); $slotCode = trim($_GET['slot_code'] ?? ''); $scope = trim($_GET['scope'] ?? ''); $stmt = getDB()->prepare( "SELECT COUNT(*) FROM kpi_entries WHERE batch_id=? AND staff_id=? AND slot_code=? AND scope=?" ); $stmt->execute([$batchId, $staffId, $slotCode, $scope]); $count = (int) $stmt->fetchColumn(); jsonOut(['exists' => $count > 0, 'count' => $count]); } /* ==================================================== Допоміжна: завантажити ваги індикаторів з БД ==================================================== */ function loadIndicatorWeights(PDO $db, string $room, string $slot, string $scope): array { $stmt = $db->prepare( "SELECT LOWER(indicator) as ind, weight FROM indicators WHERE (room_code=? OR room_code='*') AND (slot_code=? OR slot_code='*') AND scope=? AND active=1" ); $stmt->execute([$room, $slot, $scope]); $map = []; foreach ($stmt->fetchAll() as $r) { $map[$r['ind']] = (float) $r['weight']; } return $map; }

6. api/reports.php — звіти і дашборд

PHP api/reports.php — ключові SQL запити

<?php declare(strict_types=1); require_once __DIR__.'/db.php'; $action = $_GET['action'] ?? ''; match($action) { 'summary' => summary(), 'violations' => violations(), 'daily' => daily(), 'monthly' => monthly(), default => jsonError('Unknown action', 404) }; function summary(): void { $from = $_GET['from'] ?? date('Y-m-01'); $to = $_GET['to'] ?? date('Y-m-d'); $room = $_GET['room'] ?? null; $where = "entry_date BETWEEN :from AND :to"; $params = ['from' => $from, 'to' => $to]; if ($room) { $where .= " AND room_code = :room"; $params['room'] = $room; } $sql = " SELECT e.staff_id, s.staff_name, e.room_code, ROUND( SUM(CASE WHEN scope='Team' THEN earned_points ELSE 0 END) / NULLIF(SUM(CASE WHEN scope='Team' THEN possible_points ELSE 0 END),0) * 100, 2 ) AS team_kpi, ROUND( SUM(CASE WHEN scope='Individual' THEN earned_points ELSE 0 END) / NULLIF(SUM(CASE WHEN scope='Individual' THEN possible_points ELSE 0 END),0) * 100, 2 ) AS ind_kpi, SUM(CASE WHEN scope='Team' THEN possible_points ELSE 0 END) AS team_possible, SUM(CASE WHEN scope='Individual' THEN possible_points ELSE 0 END) AS ind_possible, r.is_admin FROM kpi_entries e LEFT JOIN staff s ON s.staff_id = e.staff_id LEFT JOIN rooms r ON r.room_code = e.room_code WHERE $where AND e.staff_id != 'TEAM' GROUP BY e.staff_id, e.room_code ORDER BY e.room_code, team_kpi DESC"; $stmt = getDB()->prepare($sql); $stmt->execute($params); $rows = $stmt->fetchAll(); // Рахуємо TotalKPI% в PHP (складна умова Admin + IndPossible=0) foreach ($rows as &$r) { $team = (float)($r['team_kpi'] ?? 0); $ind = (float)($r['ind_kpi'] ?? 0); $indP = (float)($r['ind_possible'] ?? 0); if ($r['is_admin'] || $indP == 0) { $r['total_kpi'] = round($team, 2); } else { $r['total_kpi'] = round($team * 0.8 + $ind * 0.2, 2); } } jsonOut(['ok' => true, 'data' => $rows]); } function violations(): void { $from = $_GET['from'] ?? date('Y-m-01'); $to = $_GET['to'] ?? date('Y-m-d'); $room = $_GET['room'] ?? null; $limit = min((int)($_GET['limit'] ?? 20), 100); $where = "check_value=0 AND entry_date BETWEEN :from AND :to"; $params = ['from'=>$from, 'to'=>$to]; if($room){ $where.=" AND room_code=:room"; $params['room']=$room; } $stmt = getDB()->prepare(" SELECT room_code, category, indicator, COUNT(*) AS violations, COUNT(DISTINCT entry_date) AS days_violated, COUNT(DISTINCT staff_id) AS staff_affected FROM kpi_entries WHERE $where AND staff_id != 'TEAM' GROUP BY room_code, category, indicator ORDER BY violations DESC LIMIT :lim"); $params['lim'] = $limit; $stmt->execute($params); jsonOut(['ok'=>true, 'data'=>$stmt->fetchAll()]); }

8. form.html + form.js

JS assets/form.js — логіка форми

// ===================================================== // Стан форми // ===================================================== const state = { config: null, // завантажується з /api/config.php batchId: null, // генерується при відкритті сторінки date: null, room: null, slot: null, staffIds: [], teamItems: [], // { indicator, category, check, comment } indItems: {} // { staffId: [items] } }; // ===================================================== // Завантаження конфігу при старті // ===================================================== async function loadConfig() { const res = await fetch('/kpi/api/config.php'); const data = await res.json(); state.config = data; renderRooms(data.rooms); renderSlots(data.slots); } // ===================================================== // Збереження — головна функція // ===================================================== async function save() { const btn = document.getElementById('btn-save'); btn.disabled = true; btn.textContent = 'Saving...'; try { // 1. Перевірка дублікату const dupUrl = `/kpi/api/kpi.php?action=checkDuplicate` + `&batch_id=${state.batchId}&staff_id=TEAM` + `&slot_code=${state.slot}&scope=Team`; const dup = await (await fetch(dupUrl)).json(); if (dup.exists) { showError('Цей слот вже збережено! Відкрийте нову форму.'); return; } // 2. Зберегти Team KPI const teamRes = await post('/kpi/api/kpi.php?action=saveTeam', { batch_id: state.batchId, date: state.date, room_code: state.room, slot_code: state.slot, present_staff_ids: state.staffIds, submitted_by: 'manager', items: state.teamItems }); if (!teamRes.ok) throw new Error(teamRes.error); // 3. Зберегти Individual KPI по кожному стафу for (const staffId of state.staffIds) { const items = state.indItems[staffId] ?? []; if (!items.length) continue; const indRes = await post('/kpi/api/kpi.php?action=saveIndividual', { batch_id: state.batchId, date: state.date, room_code: state.room, slot_code: state.slot, staff_id: staffId, submitted_by: 'manager', items: items }); if (!indRes.ok) throw new Error(indRes.error); } // 4. Успіх showSuccess('✓ KPI saved successfully'); btn.textContent = 'Saved ✓'; } catch (err) { showError('Error: ' + err.message); btn.disabled = false; btn.textContent = 'Save KPI'; } } async function post(url, data) { const res = await fetch(url, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(data) }); return res.json(); }
